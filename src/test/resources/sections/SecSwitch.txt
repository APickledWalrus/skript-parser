test:
	set {var} to "Hello"
	switch {var}:
		case "World":
			add 1 to {flag::*}
		when "Hello":
			add 2 to {flag::*}
			set {var2} to 2
			add 1 to {var2}
			add {var2} to {flag::*}
		case 2 or 3:
			add 4 to {flag::*}
		case 4 or "Hello":
			add 5 to {flag::*}
		case "Hello" then add 8 to {flag::*}
		default:
			add 6 to {flag::*}

	# 2, 3, 5 and 8
	assert {flag::*} = 2, 3, 5 and 8 with "{flag::*} should be 2, 3, 5 and 8: %{flag::*}%"

	set {var} to 3
	clear {flag::*}
	switch {var}:
		case 1:
			add 1 to {flag::*}
		when 2 then add 2 to {flag::*}
		default:
			add 3 to {flag::*}

	# 3 and 4
    assert {flag::*} = 3 with "{flag::*} should be 3: %{flag::*}%"
